120:export function evaluateAttacks(unitToAct: Unit, context: UnitActionContext): { action: any, score: number } | null {
121:  const { 
122:    state, 
123:    currentPlayer, 
124:    threatMatrix, 
125:    eminentThreatBases,
126:    leaderId, 
127:    focusOnLeader, 
128:    hvt, 
129:    isBarbarian,
130:    primaryAggressorId,
131:    unitsMap,
132:    boardMap,
133:    otherFriendlyUnits,
134:    friendlyFollowUpPotential,
135:    config
136:  } = context;
137:
138:  const attacks = getValidAttacks(unitToAct, state.board, state.units);
139:  if (attacks.length === 0) return null;
140:
141:  let bestAttack = attacks[0];
142:  let maxPriority = -Infinity;
143:
144:  const unitCurrentThreat = threatMatrix.get(`${unitToAct.coord.q},${unitToAct.coord.r}`);
145:  const isUnitInPeril = unitCurrentThreat ? unitCurrentThreat.minTurns === 1 : false;
146:  const isUnitDoomed = unitCurrentThreat ? (unitCurrentThreat.eminentThreatValue >= UNIT_STATS[unitToAct.type].cost * 1.5) : false;
147:
148:  const attackSafety = new LoopSafety('evaluateAttacks', 1000);
149:  for (const a of attacks) {
150:    if (attackSafety.tick()) break;
151:    
152:    const coordKey = `${a.q},${a.r}`;
153:    const targetUnit = unitsMap.get(coordKey);
154:    const targetTile = boardMap.get(coordKey);
155:
156:    let priority = 0;
157:
158:    const evaluation = context.evaluationMap.get(coordKey);
159:    if (evaluation) {
160:      priority += evaluation.score * 8.0;
161:    }
162:
163:    if (targetUnit) {
164:      const targetValue = UNIT_STATS[targetUnit.type].cost;
165:      // Always give a base reward for engaging enemy units
166:      // If unit is in peril or doomed, increase priority to "take them with you"
167:      const meatShieldBonus = (isUnitInPeril ? 1.5 : 1.0) * (isUnitDoomed ? 2.0 : 1.0);
168:      priority = (targetValue + config.BASE_REWARD * 20.0) * meatShieldBonus;
169:      
170:      const potentialAttackers = otherFriendlyUnits.filter(u => {
171:        if (u.hasActed) return false;
172:        const dist = getDistance(u.coord, targetUnit.coord, state.board);
173:        return dist <= getUnitRange(u, state.board);
174:      });
175:      
176:      if (potentialAttackers.length > 0) {
177:        priority += config.BASE_REWARD * FOCUS_FIRE_BONUS * potentialAttackers.length;
178:      }
179:
180:      if (potentialAttackers.length >= 1) {
181:        priority += config.BASE_REWARD * config.KILL_PRIORITY_BONUS;
182:      }
183:
184:      // LETHALITY: Huge bonus for finishing off a unit that is threatening a base
185:      const isTargetThreateningBase = eminentThreatBases.some(b => {
186:        const distToBase = getDistance(targetUnit.coord, b.coord, state.board);
187:        const enemyRange = UNIT_STATS[targetUnit.type].moves + getUnitRange(targetUnit, state.board);
188:        return distToBase <= enemyRange;
189:      });
190:
191:      if (isTargetThreateningBase) {
192:        priority += config.BASE_REWARD * config.LETHALITY_WEIGHT_BONUS; // Urgent defense priority
193:      }
194:      
195:      if (isUnitInPeril && !isBarbarian) {
196:        priority += config.BASE_REWARD * 10.0;
197:        const targetThreatRadius = UNIT_STATS[targetUnit.type].moves + getUnitRange(targetUnit, state.board);
198:        if (getDistance(unitToAct.coord, targetUnit.coord, state.board) <= targetThreatRadius) {
199:          priority += config.BASE_REWARD * 5.0;
200:        }
201:      }
202:
203:      if (hvt && targetUnit.id === hvt.id) priority += config.BASE_REWARD * 3.0;
204:      if (targetUnit.type === UnitType.CATAPULT) priority += config.BASE_REWARD * 1.0;
205:      if (focusOnLeader && targetUnit.ownerId === leaderId) priority += config.BASE_REWARD * 0.5;
206:      if (primaryAggressorId !== -1 && targetUnit.ownerId === primaryAggressorId) priority += config.BASE_REWARD * 15.0;
207:
208:      if (unitCurrentThreat && !isBarbarian) {
209:         const targetThreatRadius = UNIT_STATS[targetUnit.type].moves + getUnitRange(targetUnit, state.board);
210:         const isTargetThreateningUs = getDistance(unitToAct.coord, targetUnit.coord, state.board) <= targetThreatRadius;
211:         
212:         // In a one-hit-kill game, if there is ANY attacker that can hit us (besides maybe the one we're killing), we are doomed.
213:         const willDieAnyway = unitCurrentThreat.eminentAttackerCount > (isTargetThreateningUs ? 1 : 0);
214:         
215:         if (willDieAnyway) {
216:            // If we're going to die anyway, any damage we deal is a bonus.
217:            // Don't penalize for "trading down" since the alternative is dying for free.
218:            priority += targetValue * 2.0 + BASE_REWARD * 2.0;
219:            
220:            // If we can take a Catapult or Knight with us, that's a massive win.
221:            if (targetValue >= 150) priority += BASE_REWARD * 10.0;
222:         }
223:      }
224:
225:      const isOnSettlement = targetTile && (targetTile.terrain === TerrainType.VILLAGE || targetTile.terrain === TerrainType.FORTRESS || targetTile.terrain === TerrainType.CASTLE || targetTile.terrain === TerrainType.GOLD_MINE);
226:      if (isOnSettlement) {
227:        const canBuddiesNeutralize = friendlyFollowUpPotential.some(f => f.attacks.some((at: any) => at.q === targetTile.coord.q && at.r === targetTile.coord.r));
228:        const canBuddiesClaim = otherFriendlyUnits.some(f => {
229:          if (f.hasActed) return false;
230:          return getDistance(f.coord, targetTile.coord, state.board) <= UNIT_STATS[f.type].moves;
231:        });
232:
233:        if (canBuddiesNeutralize && canBuddiesClaim && targetTile.ownerId !== currentPlayer.id) {
234:          priority += BASE_REWARD * COMBO_FOLLOW_UP_NEUTRALIZER_BONUS;
235:        }
236:      }
237:      
238:      if (unitToAct.type === UnitType.INFANTRY && isOnSettlement) priority += BASE_REWARD * INFANTRY_VANGUARD_SETTLEMENT_BONUS;
239:
240:      const isThreateningBase = context.eminentThreatBases.some(b => getDistance(a, b.coord, state.board) <= 3);
241:      if (isThreateningBase) priority += BASE_REWARD * PREEMPTIVE_DEFENSE_BONUS;
242:
243:      const isNearOurSettlement = context.mySettlements.some(s => getDistance(a, s.coord, state.board) <= 2);
244:      if (isNearOurSettlement) priority += BASE_REWARD * DRIVE_OUT_BONUS;
245:
246:      const isOccupyingMySettlement = targetTile && targetTile.ownerId === currentPlayer.id && (targetTile.terrain === TerrainType.VILLAGE || targetTile.terrain === TerrainType.FORTRESS || targetTile.terrain === TerrainType.CASTLE || targetTile.terrain === TerrainType.GOLD_MINE);
247:      if (isOccupyingMySettlement) priority += BASE_REWARD * 2.0;
248:
249:      // --- NEW: Target Rarity and Coordination ---
250:      // "coordinate attacks to hit the most number of enemy units as possible per turn"
251:      const othersWhoCanHitThis = friendlyFollowUpPotential.filter(f => f.id !== unitToAct.id && f.attacks.some((at: any) => at.q === a.q && at.r === a.r));
252:      if (othersWhoCanHitThis.length === 0) {
253:        // I am the ONLY unit that can hit this target. Definitely prioritize it!
254:        priority += config.BASE_REWARD * 20.0;
255:      } else {
256:        // Others can hit it too. Check if they have other options.
257:        // If all others who can hit this have NO other targets, I should probably leave it for them if I have other targets.
258:        const myOtherTargets = attacks.length - 1;
259:        const othersTargetsCounts = othersWhoCanHitThis.map(f => f.attacks.length);
260:        const othersAreRestricted = othersTargetsCounts.every(count => count === 1);
261:        
262:        if (othersAreRestricted && myOtherTargets > 0) {
263:          // I have other options, and my buddies only have THIS one. I should wait/pick something else.
264:          priority -= config.BASE_REWARD * 15.0;
265:        }
266:      }
267:
268:      // SIEGE PRIORITY: Catapults should focus on cracking enemy fortifications
269:      if (unitToAct.type === UnitType.CATAPULT && targetTile && targetTile.ownerId !== currentPlayer.id && (targetTile.terrain === TerrainType.FORTRESS || targetTile.terrain === TerrainType.CASTLE)) {
270:        priority += config.BASE_REWARD * config.SIEGE_OVERRIDE_BONUS; 
271:      }
272:
273:    } else if (targetTile && targetTile.ownerId !== null && targetTile.ownerId !== currentPlayer.id && (targetTile.terrain === TerrainType.VILLAGE || targetTile.terrain === TerrainType.FORTRESS || targetTile.terrain === TerrainType.CASTLE || targetTile.terrain === TerrainType.GOLD_MINE)) {
274:      const settlementValue = SETTLEMENT_INCOME[targetTile.terrain as TerrainType] * HORIZON;
275:      priority = settlementValue * 1.5 + BASE_REWARD * 4.0;
276:      
277:      const enemyUnitsNearThisSettlement = state.units.filter(u => u.ownerId === targetTile.ownerId && getDistance(u.coord, targetTile.coord, state.board) <= UNIT_STATS[u.type].moves);
278:      if (enemyUnitsNearThisSettlement.length > 0) priority += BASE_REWARD * SETTLEMENT_DEGRADATION_PRIORITY_BONUS;
279:
280:      const canBuddiesClaim = otherFriendlyUnits.some(f => {
281:          if (f.hasActed) return false;
282:          return getDistance(f.coord, targetTile.coord, state.board) <= UNIT_STATS[f.type].moves;
283:      });
284:      
285:      if (canBuddiesClaim && targetTile.terrain === TerrainType.VILLAGE) {
286:        priority += BASE_REWARD * COMBO_FOLLOW_UP_CLAIMER_BONUS;
287:      } else if (canBuddiesClaim) {
288:        priority += BASE_REWARD * (COMBO_FOLLOW_UP_CLAIMER_BONUS * 0.5);
289:      }
290:
291:      if (isBarbarian) priority += BASE_REWARD * PILLAGE_SCORE_BONUS;
292:      if (isUnitInPeril && !isBarbarian) priority += BASE_REWARD * 5.0;
293:      if (focusOnLeader && targetTile.ownerId === leaderId) priority += BASE_REWARD * 1.0;
294:      if (primaryAggressorId !== -1 && targetTile.ownerId === primaryAggressorId) priority += BASE_REWARD * 10.0;
295:    }
296:
297:    if (priority > maxPriority) {
298:      // Tactical Lookahead for high-value targets or high-value units
299:      if ((unitToAct.type === UnitType.KNIGHT || unitToAct.type === UnitType.CATAPULT) && !isBarbarian) {
300:        const safety = evaluateActionSafety(state, { type: 'attack', payload: { unitId: unitToAct.id, target: a } }, currentPlayer.id);
301:        if (!safety.isSafe) {
302:          // Penalize unsafe attacks for high value units
303:          priority -= (UNIT_STATS[unitToAct.type].cost * 2.0); 
304:        }
305:      }
306:
307:      if (priority > maxPriority) {
308:        maxPriority = priority;
309:        bestAttack = a;
310:      }
311:    }
312:  }
313:
314:  if (maxPriority > -100 || (isBarbarian && attacks.length > 0)) {
315:    return { 
316:      action: { type: 'attack' as const, payload: { unitId: unitToAct.id, target: bestAttack } },
317:      score: maxPriority === -Infinity ? 0 : maxPriority
318:    };
319:  }
320:  return null;
321:}

323:export function evaluateMoves(unitToAct: Unit, context: UnitActionContext): { action: any, score: number } | null {
324:  const {
325:    state,
326:    currentPlayer,
327:    threatMatrix,
328:    influenceMap,
329:    eminentThreatBases,
330:    empireCenter,
331:    hvt,
332:    isSavingForMine,
333:    isSavingForVillage,
334:    isCriticallyLaggingLargeEconomy,
335:    isBarbarian,
336:    cachedData,
337:    boardMap,
338:    mySettlements,
339:    enemySettlements,
340:    myCatapults,
341:    otherFriendlyUnits,
342:    friendlyFollowUpPotential,
343:    globalAggression,
344:    globalUnitRatio,
345:    isRich,
346:    unitsMap,
347:    config
348:  } = context;
349:
350:  const moves = getValidMoves(unitToAct, state.board, state.units);
351:  moves.push(unitToAct.coord);
352:  
353:  if (moves.length === 0) return null;
354:
355:  let bestMove = moves[0];
356:  let maxScore = -Infinity;
357:
358:  const moveSafety = new LoopSafety('evaluateMoves', 5000);
359:  
360:  if (!cachedData.settlementTargets) {
361:    cachedData.settlementTargets = state.board.filter(t => 
362:      (t.ownerId === null || t.ownerId !== currentPlayer.id) && 
363:      (t.terrain === TerrainType.VILLAGE || t.terrain === TerrainType.FORTRESS || t.terrain === TerrainType.CASTLE || t.terrain === TerrainType.GOLD_MINE)
364:    ).map(t => ({ coord: t.coord, value: SETTLEMENT_INCOME[t.terrain as TerrainType] * HORIZON, isExpansionHub: false, isSettlement: true, ownerId: t.ownerId, unitType: undefined }));
365:  }
366:  const settlementTargets = cachedData.settlementTargets;
367:
368:  if (!cachedData.enemyUnitTargets) {
369:    cachedData.enemyUnitTargets = state.units.filter(u => u.ownerId !== currentPlayer.id)
370:      .map(u => ({ coord: u.coord, value: UNIT_STATS[u.type].cost, isExpansionHub: false, isSettlement: false, ownerId: u.ownerId, unitType: u.type }));
371:  }
372:  const enemyUnitTargets = cachedData.enemyUnitTargets;
373:
374:  if (!cachedData.expansionHubs) {
375:    cachedData.expansionHubs = state.board.filter(t => 
376:      t.terrain === TerrainType.PLAINS && t.ownerId === null &&
377:      !state.units.some(u => u.ownerId === currentPlayer.id && getDistance(u.coord, t.coord, state.board) <= 3) &&
378:      !mySettlements.some(s => getDistance(s.coord, t.coord, state.board) <= 4)
379:    ).map(t => {
380:      const neighbors = _getNeighbors(t.coord);
381:      let quality = 1;
382:      neighbors.forEach(n => {
383:        const nt = boardMap.get(`${n.q},${n.r}`);
384:        if (nt && nt.ownerId === null) quality++;
385:      });
386:      return { coord: t.coord, value: quality * 20, isExpansionHub: true, isSettlement: false, ownerId: null, unitType: undefined };
387:    }).sort((a, b) => b.value - a.value).slice(0, 5);
388:  }
389:  const expansionHubs = cachedData.expansionHubs;
390:
391:  const moveTargets = [
392:    ...settlementTargets,
393:    ...enemyUnitTargets,
394:    ...expansionHubs
395:  ].sort((a, b) => getDistance(unitToAct.coord, a.coord, state.board) - getDistance(unitToAct.coord, b.coord, state.board))
396:   .slice(0, 30);
397:
398:  const stats = UNIT_STATS[unitToAct.type];
399:  const myValue = stats.cost;
400:  const currentThreat = threatMatrix.get(`${unitToAct.coord.q},${unitToAct.coord.r}`);
401:  const isUnitInPeril = currentThreat ? currentThreat.minTurns === 1 : false;
402:
403:  const hasSafeRetreat = isUnitInPeril && moves.some(m => {
404:    const t = threatMatrix.get(`${m.q},${m.r}`);
405:    return !t || t.minTurns > 1;
406:  });
407:
408:  for (const m of moves) {
409:    if (moveSafety.tick()) break;
410:    
411:    let score = 0;
412:    const tile = boardMap.get(`${m.q},${m.r}`)!;
413:    const isStayPut = m.q === unitToAct.coord.q && m.r === unitToAct.coord.r;
414:    
415:    const moveThreat = threatMatrix.get(`${m.q},${m.r}`);
416:    const moveThreatLevel = moveThreat ? moveThreat.minTurns : Infinity;
417:    const isMoveInPeril = moveThreatLevel === 1;
418:
419:    const evaluation = context.evaluationMap.get(`${m.q},${m.r}`);
420:    if (evaluation) {
421:      score += evaluation.score * 8.0; // Increased significantly to respect the global Strategic Analysis
422:    }
423:
424:    if (isStayPut) {
425:      score += BASE_REWARD * STAY_PUT_BIAS;
426:      const minDistToMySettlement = mySettlements.length > 0 ? Math.min(...mySettlements.map(s => getDistance(m, s.coord, state.board))) : 0;
427:      if (minDistToMySettlement >= stats.moves && tile.terrain === TerrainType.PLAINS && tile.ownerId === null) {
428:        score += BASE_REWARD * 10.0;
429:      }
430:      const targetSettlementBeingNetted = moveTargets.some(t => 
431:        t.isSettlement && t.ownerId !== null && t.ownerId !== currentPlayer.id &&
432:        friendlyFollowUpPotential.some(f => f.attacks.some((at: any) => at.q === t.coord.q && at.r === t.coord.r))
433:      );
434:      if (targetSettlementBeingNetted) score -= BASE_REWARD * COMBO_SEQUENCING_STAY_PUT_PENALTY;
435:    }
436:
437:    if (!isMoveInPeril && tile.ownerId === null && 
438:       (tile.terrain === TerrainType.VILLAGE || tile.terrain === TerrainType.FORTRESS || tile.terrain === TerrainType.CASTLE || tile.terrain === TerrainType.GOLD_MINE)) {
439:      score += (BASE_REWARD * 30.0);
440:    }
441:
442:    if (tile.terrain === TerrainType.MOUNTAIN && isSavingForMine) {
443:      const myIncome = mySettlements.reduce((sum, s) => sum + SETTLEMENT_INCOME[s.terrain as TerrainType], 0);
444:      const goldNextTurn = currentPlayer.gold + myIncome;
445:      const mineCost = _UPGRADE_COSTS[TerrainType.GOLD_MINE];
446:      if (goldNextTurn < mineCost) score -= (BASE_REWARD * 15.0); 
447:      else if (goldNextTurn >= mineCost) score += (BASE_REWARD * 20.0);
448:    }
449:
450:    if (tile.ownerId === null && (tile.terrain === TerrainType.VILLAGE || tile.terrain === TerrainType.FORTRESS || tile.terrain === TerrainType.CASTLE || tile.terrain === TerrainType.GOLD_MINE)) {
451:      score += (SETTLEMENT_INCOME[tile.terrain as TerrainType] * HORIZON) + BASE_REWARD * IMMEDIATE_CAPTURE_BONUS; 
452:    }
453:
454:    const potentialRange = getUnitRange({ ...unitToAct, coord: m }, state.board);
455:
456:    if (!isMoveInPeril || isBarbarian) {
457:      const perilSafety = new LoopSafety('evaluateMoves-peril', 100);
458:      let putsEnemyInPeril = false;
459:      let perilCount = 0;
460:      for (const target of moveTargets) {
461:        if (perilSafety.tick()) break;
462:        const dist = getDistance(m, target.coord, state.board);
